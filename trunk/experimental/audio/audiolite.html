SWFAudioLite Test using niftyplayer.swf

复杂多功能使用可用 swfaudio.js，都具有 play（开始/继续）,stop（停止）,pause（暂停）

代码：

	KISSY.ready(function(S){
	/**
		
		指定本地的swf地址
	*/
	var sm = new S.SWFAudioLite("niftyplayer.swf");
	
	sm.on("ready",function(){
		//立即开始播放
		sm.loadAndPlay("waka.mp3");
		
		//停止
		var _stop=S.one("#_stop");
		_stop.on("click",function(){
			sm.stop();
		});
		
		//开始/继续当前
		var _start=S.one("#_start");
		_start.on("click",function(){
			sm.play();
		});
		
		//暂停
		var _pause=S.one("#_pause");
		_pause.on("click",function(){
			sm.pause();
		});
	
		
	});

});
